80 {
81 static_assert(TIsDerivedFrom<ActivatableWidgetT, UCommonActivatableWidget>::IsDerived, "Only CommonActivatableWidgets can be used here");
82
83 static FName NAME_PushingWidgetToLayer("PushingWidgetToLayer");
84 const FName SuspendInputToken = bSuspendInputUntilComplete ? UGameplayCommonUILibrary::SuspendInputForPlayer(GetOwningPlayer(), NAME_PushingWidgetToLayer) : NAME_None;
85
86 FStreamableManager& StreamableManager = UAssetManager::Get().GetStreamableManager();
87 TSharedPtr<FStreamableHandle> StreamingHandle = StreamableManager.RequestAsyncLoad(ActivatableWidgetClass.ToSoftObjectPath(), FStreamableDelegate::CreateWeakLambda(this,
88 [this, LayerTag, ActivatableWidgetClass, StateFunc, SuspendInputToken]()
89 {
90 UGameplayCommonUILibrary::ResumeInputForPlayer(GetOwningPlayer(), SuspendInputToken);
91
92 ActivatableWidgetT* Widget = PushWidgetToLayerStack<ActivatableWidgetT>(LayerTag, ActivatableWidgetClass.Get(), [StateFunc](ActivatableWidgetT& WidgetToInit)
93 {
94 StateFunc(EGameplayWidgetLayerAsyncState::Initialize, &WidgetToInit);
95 });
96
97 StateFunc(EGameplayWidgetLayerAsyncState::AfterPush, Widget);
98 Widget->SetFocus();
99 }));
100
101 // Set up a cancel delegate so that we can resume input if this handler is canceled.
102 StreamingHandle->BindCancelDelegate(FStreamableDelegate::CreateWeakLambda(this, [this, StateFunc, SuspendInputToken]()
103 {
104 UGameplayCommonUILibrary::ResumeInputForPlayer(GetOwningPlayer(), SuspendInputToken);
105 StateFunc(EGameplayWidgetLayerAsyncState::Canceled, nullptr);
106 }));
107
108 return StreamingHandle;

119 template <typename ActivatableWidgetT = UCommonActivatableWidget>
120 ActivatableWidgetT* PushWidgetToLayerStack(FGameplayTag LayerTag, UClass* ActivatableWidgetClass, TFunctionRef<void(ActivatableWidgetT&)> InitInstanceFunc)
121 {
122 static_assert(TIsDerivedFrom<ActivatableWidgetT, UCommonActivatableWidget>::IsDerived, "Only CommonActivatableWidgets can be used here");
123
124 if (UCommonActivatableWidgetContainerBase* Layer = GetLayerStackContainer(LayerTag))
125 {
126 UE_LOG(LogGameplayPrimaryLayout, Display, TEXT("Pushing Widget [%s] to Layer [%s]"), *GetNameSafe(ActivatableWidgetClass), *LayerTag.ToString());
127
128 return Layer->AddWidget<ActivatableWidgetT>(ActivatableWidgetClass, InitInstanceFunc);
129 }
130
131 UE_LOG(LogGameplayPrimaryLayout, Warning, TEXT("Failed to push widget [%s]: Layer [%s] not found!"), *GetNameSafe(ActivatableWidgetClass), *LayerTag.ToString());
132 return nullptr;
133 }
