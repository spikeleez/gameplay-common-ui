53 void RecordSample(const double Sample)
54 {

101 void ForEachCurrentSample(const TFunctionRef<void(const double Stat)> Func) const
102 {
103 int32 Index = CurrentSampleIndex;
104 for (int32 i = 0; i < SampleSize; i++)
105 {
106 Func(Samples[Index]);
107
108 Index++;
109 if (Index == SampleSize)
110 {
111 Index = 0;
112 }
113 }
114 }

120 int32 GetSampleSize() const

123 }
124
129 double GetAverage() const
130 {
131 double Sum = 0.0;
132 ForEachCurrentSample([&Sum](const double Stat)

137 // Return the average stat value.
