118class GAMEPLAYCOMMONSETTINGS_API FGameplaySettingEditCondition_FramePacingMode final : public FGameplaySettingEditCondition
119{
120public:
126 FGameplaySettingEditCondition_FramePacingMode(EGameplayFramePacingMode InDesiredMode, EGameplayFramePacingEditCondition InMatchMode = EGameplayFramePacingEditCondition::EnableIf)
127 : DesiredMode(InDesiredMode)
128 , MatchMode(InMatchMode)
129 {}
130
131 virtual void GatherEditState(const ULocalPlayer* InLocalPlayer, FGameplaySettingEditableState& InOutEditState) const override
132 {
133 const EGameplayFramePacingMode ActualMode = UGameplayPlatformSpecificRenderingSettings::Get()->FramePacingMode;
134
135 const bool bMatches = (ActualMode == DesiredMode);
136 const bool bMatchesAreBad = (MatchMode == EGameplayFramePacingEditCondition::DisableIf);
137
138 if (bMatches == bMatchesAreBad)
139 {
140 InOutEditState.Kill(FString::Printf(TEXT("Frame pacing mode %d didn't match requirement %d"), (int32)ActualMode, (int32)DesiredMode));
141 }
142 }
143private:
145 EGameplayFramePacingMode DesiredMode;
147 EGameplayFramePacingEditCondition MatchMode;
148};

212class GAMEPLAYCOMMONSETTINGS_API FGameplaySettingEditCondition_LatencyStatsCurrentlyEnabled final : public FGameplaySettingEditCondition
213{
214public:
215 FGameplaySettingEditCondition_LatencyStatsCurrentlyEnabled() = default;
216
217 virtual ~FGameplaySettingEditCondition_LatencyStatsCurrentlyEnabled() override
218 {
219 if (!SettingChangedDelegate.IsValid()) return;
220
221 UGameplaySettingsLocal* Settings = UGameplaySettingsLocal::Get();
222 if (!Settings) return;
223
224 Settings->OnLatencyStatIndicatorSettingsChangedEvent().Remove(SettingChangedDelegate);
225 }
226
227private:
228 virtual void Initialize(const ULocalPlayer* InLocalPlayer) override
229 {
230 // Bind to an event for when the settings are updated so that we can broadcast that we need
231 // to be re-evaluated
232 UGameplaySettingsLocal* Settings = UGameplaySettingsLocal::Get();
233 if (!Settings) return;
234
235 SettingChangedDelegate = Settings->OnLatencyStatIndicatorSettingsChangedEvent().AddSP(this->AsShared(),
236 &FGameplaySettingEditCondition_LatencyStatsCurrentlyEnabled::BroadcastEditConditionChanged);
237 }
238
239 virtual void GatherEditState(const ULocalPlayer* InLocalPlayer, FGameplaySettingEditableState& InOutEditState) const override
240 {
241 const UGameplaySettingsLocal* Settings = UGameplaySettingsLocal::Get();
242 if (Settings && !Settings->GetEnableLatencyTrackingStats())
243 {
244 InOutEditState.Disable(LOCTEXT("LatencyMarkerRequireStatsEnabled", "Latency Tracking Stats must be enabled to use this."));
245 }
246 }
247
248private:
250 FDelegateHandle SettingChangedDelegate;
251};

274 }
